126 def test_get_sysctl_missing_binary(self): 

127 module = MagicMock() 

128 module.get_bin_path.return_value = '/usr/sbin/sysctl' 

129 module.run_command.side_effect = ValueError 

130 self.assertRaises(ValueError, get_sysctl, module, ['vm'])

132 def test_get_sysctl_nonzero_rc(self): 

133 module = MagicMock() 

134 module.get_bin_path.return_value = '/usr/sbin/sysctl' 

135 module.run_command.return_value = (1, '', '') 

136 sysctl = get_sysctl(module, ['hw']) 

137 self.assertEqual(sysctl, {})

139 def test_get_sysctl_command_error(self): 

140 module = MagicMock() 

141 module.get_bin_path.return_value = '/usr/sbin/sysctl' 

142 for err in (IOError, OSError): 

143 module.reset_mock() 

144 module.run_command.side_effect = err('foo') 

145 sysctl = get_sysctl(module, ['hw']) 

146 module.warn.assert_called_once_with('Unable to read sysctl: foo') 

147 self.assertEqual(sysctl, {}) 

148 

149 def test_get_sysctl_all_invalid_output(self): 

150 module = MagicMock() 

151 module.get_bin_path.return_value = '/sbin/sysctl' 

152 module.run_command.return_value = (0, BAD_SYSCTL, '') 

153 sysctl = get_sysctl(module, ['hw']) 

154 module.run_command.assert_called_once_with(['/sbin/sysctl', 'hw']) 

155 lines = [l for l in BAD_SYSCTL.splitlines() if l] 

156 for call in module.warn.call_args_list: 

157 self.assertIn('Unable to split sysctl line', call[0][0]) 

158 self.assertEqual(module.warn.call_count, len(lines)) 

159 self.assertEqual(sysctl, {}) 

160 

161 def test_get_sysctl_mixed_invalid_output(self): 

162 module = MagicMock() 

163 module.get_bin_path.return_value = '/sbin/sysctl' 

164 module.run_command.return_value = (0, GOOD_BAD_SYSCTL, '') 

165 sysctl = get_sysctl(module, ['hw']) 

166 module.run_command.assert_called_once_with(['/sbin/sysctl', 'hw']) 

167 bad_lines = ['bad.output.here', 'and.bad.output.here'] 

168 for call in module.warn.call_args_list: 

169 self.assertIn('Unable to split sysctl line', call[0][0]) 

170 self.assertEqual(module.warn.call_count, 2) 

171 self.assertEqual(sysctl, {'hw.smt': '0'})

173 def test_get_sysctl_openbsd_hw(self): 

174 expected_lines = [l for l in OPENBSD_SYSCTL_HW.splitlines() if l] 

175 module = MagicMock() 

176 module.get_bin_path.return_value = '/sbin/sysctl' 

177 module.run_command.return_value = (0, OPENBSD_SYSCTL_HW, '')

182 self.assertEqual(sysctl['hw.smt'], '0') # random line

184 # weird chars in value 

185 self.assertEqual( 

186 sysctl['hw.disknames'], 

187 'cd0:,sd0:9e1bd96cb20ab429,fd0:') 

188 # more symbols/spaces in value 

189 self.assertEqual( 

190 sysctl['hw.product'], 

191 'Standard PC (i440FX + PIIX, 1996)')
