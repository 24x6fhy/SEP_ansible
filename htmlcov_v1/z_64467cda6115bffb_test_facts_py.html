35class BaseTestFactsPlatform(unittest.TestCase): 

36 platform_id = 'Generic' 

37 fact_class = hardware.base.Hardware 

38 collector_class = None 

39 

40 """Verify that the automagic in Hardware.__new__ selects the right subclass.""" 

41 @patch('platform.system') 

42 def test_new(self, mock_platform): 

43 if not self.fact_class: 

44 pytest.skip('This platform (%s) does not have a fact_class.' % self.platform_id) 

45 mock_platform.return_value = self.platform_id 

46 inst = self.fact_class(module=Mock(), load_on_init=False) 

47 self.assertIsInstance(inst, self.fact_class) 

48 self.assertEqual(inst.platform, self.platform_id) 

49 

50 def test_subclass(self): 

51 if not self.fact_class: 

52 pytest.skip('This platform (%s) does not have a fact_class.' % self.platform_id) 

53 # 'Generic' will try to map to platform.system() that we are not mocking here 

54 if self.platform_id == 'Generic': 

55 return 

56 inst = self.fact_class(module=Mock(), load_on_init=False) 

57 self.assertIsInstance(inst, self.fact_class) 

58 self.assertEqual(inst.platform, self.platform_id) 

59 

60 def test_collector(self): 

61 if not self.collector_class: 

62 pytest.skip('This test class needs to be updated to specify collector_class') 

63 inst = self.collector_class() 

64 self.assertIsInstance(inst, self.collector_class) 

65 self.assertEqual(inst._platform, self.platform_id)
